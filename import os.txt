def init_database():
    """Khởi tạo cấu trúc bảng dựa trên schema của bạn"""
    print("[-] Đang kết nối và khởi tạo cấu trúc bảng trên cloud...")
    
    # Thêm lệnh DROP TABLE ở đầu để làm sạch cấu trúc cũ, giữ nguyên comment của bạn
    schema_sql = """
    DROP TABLE IF EXISTS specs_tws CASCADE;
    DROP TABLE IF EXISTS specs_wired CASCADE;
    DROP TABLE IF EXISTS specs_speakers CASCADE;
    DROP TABLE IF EXISTS audio_products CASCADE;
    DROP TABLE IF EXISTS categories CASCADE;

    -- 1. Bảng danh mục để phân loại thiết bị
    CREATE TABLE categories (
        id SERIAL PRIMARY KEY,
        name VARCHAR(50) NOT NULL UNIQUE -- Thêm UNIQUE để tránh trùng lặp danh mục
    );

    -- 2. Bảng sản phẩm gốc (Chứa thông tin chung)
    CREATE TABLE audio_products (
        id SERIAL PRIMARY KEY,
        category_id INT REFERENCES categories(id),
        model_name VARCHAR(100) NOT NULL,
        brand VARCHAR(50) NOT NULL,
        price_vnd INT NOT NULL,
        sound_signature VARCHAR(30), -- Bass-heavy, Neutral, V-Shape, Warm
        review_summary TEXT
    );

    -- 3. Bảng thông số riêng cho Tai nghe không dây (TWS)
    CREATE TABLE specs_tws (
        product_id INT PRIMARY KEY REFERENCES audio_products(id) ON DELETE CASCADE,
        anc_depth_db INT DEFAULT 0,
        battery_life_total INT,
        codec_score INT -- SBC=1, AAC=2, aptX=3, LDAC=5
    );

    -- 4. Bảng thông số riêng cho Tai nghe có dây (Wired)
    CREATE TABLE specs_wired (
        product_id INT PRIMARY KEY REFERENCES audio_products(id) ON DELETE CASCADE,
        impedance_ohm INT, -- Trở kháng (Ví dụ: 16, 32, 80 Ohm)
        sensitivity_db INT, -- Độ nhạy
        connector_type VARCHAR(20) -- 3.5mm, Type-C, 2-pin, MMCX
    );

    -- 5. Bảng thông số riêng cho Loa (Speakers)
    CREATE TABLE specs_speakers (
        product_id INT PRIMARY KEY REFERENCES audio_products(id) ON DELETE CASCADE,
        power_watts INT, -- Công suất (Ví dụ: 20W, 40W, 100W)
        connection_type VARCHAR(50), -- Bluetooth, AUX, Optical, RCA
        speaker_type VARCHAR(30) -- Portable (Di động), Bookshelf, Soundbar
    );
    """
    
    try:
        conn = psycopg2.connect(DATABASE_URL)
        cur = conn.cursor()
        cur.execute(schema_sql)
        conn.commit()
        cur.close()
        conn.close()
        print("[✓] Khởi tạo cấu trúc bảng thành công!")
    except Exception as e:
        print(f"[X] Lỗi khởi tạo Database: {e}")

def clean_val(val, target_type=str):
    """Hàm bổ trợ để dọn dẹp giá trị NaN từ Pandas sang None (NULL trong SQL)"""
    if pd.isna(val) or val == "" or (isinstance(val, float) and math.isnan(val)):
        return None
    try:
        return target_type(val)
    except:
        return None
